### Props
| Property | Type  | Values | Default  |
| ------ | --------- |------ | --------- |
| hrefUrl      | string   |  | |
| hrefTarget    | string   |  <code>_blank</code><code>_self</code><code>_parent</code><code>_top</code><code>framename</code>  | <code>_self</code>  |
| hrefClass    | string   |  |  |

------

export const Template = (args) => ({
    component: HrefComponent,
    props: {
        ...args,
        hrefTarget: '_self'

    moduleMetadata: {

    template: '<app-href ' +

        '[hrefUrl]="hrefUrl" ' +

})

# Usages

```typescript
@NgModule({
    imports: [LinksModule]
})

export class YourModule {}
```
