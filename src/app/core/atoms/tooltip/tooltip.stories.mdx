# Tag
<br/>

### Props
| Property | Type  | Values | Default  |
| ------ | --------- |------ | --------- |
| position    | string |  <code>left</code><code>right</code><code>above</code><code>below</code><code>before</code><code>after</code> | <code>right</code> |
| touchGestures    | string |  <code>auto</code><code>on</code><code>off</code> |  |
| value    | string |   | "tooltip" |
| innerClass    | string |   | "" |
| showDelay    | number |   |  |
| hideDelay    | number |   |  |
| disabled    | boolean |  true/false       | false  |


export const Template = (args) => ({
    component: TooltipComponent,
    props: {
        value: 'tooltip',
        innerClass: '',
        position: 'right',
        ...args
    },
    moduleMetadata: {
        imports: [MatTooltipModule, BrowserAnimationsModule],
        declarations: [TooltipComponent],
    },
    template: '<app-tooltip ' +
        '[value]="value" ' +
        '[position]="position" ' +
        '[touchGestures]="touchGestures" ' +
        '[showDelay]="showDelay" ' +
        '[hideDelay]="hideDelay" ' +
        '[disabled]="disabled" ' +
        '[innerClass]="innerClass">{{content}}</app-tooltip>',
})

# Usages

Import InputsModule into your custom module

```typescript
import { TooltipModule } from '@atoms/tooltip/tooltip.module';

@NgModule({
    imports: [TooltipModule]
})

export class YourModule {}
```

## Examples

  ```html
    <app-tooltip value="3 seconds" hideDelay="2000">2 seconds</app-tooltip>
```
 ```html
    <app-tooltip value="Above" position="above">Above</app-tooltip>
  ```
